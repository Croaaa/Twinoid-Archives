var x = function () {
 var g = $('#ghost_pages'), 
 d = g.find('.document'), z = '',
 h = window.location.hash || z,
 bkey = new URLSearchParams(h.substring(h.lastIndexOf('?'))
 ).get('bkey').split(';')[0] || z,
 title = g.find('h2').html() || z,
 cls = d.attr('class') || z,
 txt = d.find('.content').html().replace(/'/gi, "\\'"),
 pages = g.find('div.pages').html() || z,
 author = g.find('div.author').html() || z,
 body = [
 'Title: ', title, '\n', 'bkey: ', bkey, '\n', cls, 
'\n', txt, '\n', author, pages
 ].join(z);
 console.log(body);
};
