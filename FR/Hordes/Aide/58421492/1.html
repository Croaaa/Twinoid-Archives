Le voilà :

function displayTooltip() {
 var who = document.getElementsByClassName("who")[0];

 if (who != undefined) {
 var names = who.getElementsByClassName("name");
 
 for (var i = 0; i < names.length; i++) {
 var name = names[i];
 var userLink = name.getElementsByClassName("userLink")[0];
 var tid_user = name.getElementsByClassName("tid_user")[0];
 
 if (userLink != undefined && tid_user != undefined) {
 userLink.appendChild(tid_user);
 }
 }
 }
}

document.getElementById("body").onload = displayTooltip;
