Essaye la solution d'Angelisium, en exécutant le code ci-dessous dans la console :

async function newAvatar(_url) {
 let hchk = await fetch("https://twinoid.com/user/account/options").then(r => r.text()),
 __dp = new DOMParser(),
 html = __dp.parseFromString(hchk, 'text/html'),
 _chk = html.querySelector("#content .options input").value;

 if(_url.startsWith('https://imgup.motion-twin.com')) {
 _url = _url.slice(29);
 }
 
 await fetch(`https://twinoid.com/user/account/avatar?chk=${_chk};avatar=${_url}`);
 window.location.reload();
}

newAvatar('https://imgup.motion-twin.com/hordes/7/c/88eca3ac_708205.jpg');
