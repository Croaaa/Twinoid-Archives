Explained:

jQuery( '#citizen_overview_table td.name p' )

is really a short-cut using CSS selector for the following code: 

// Find the TAB with ID #citizen_overview_table
jQuery( '#citizen_overview_table' )

// Inside find a table entry tag with the name CSS class
.find( 'td.name' )

// Inside find a paragraph tag
.find( 'p' )

and here's the full snippet explained:

// Declare a text empty string variable 't'
var t=''; 

// Inside the Citizen Overview tab
// Find the table entry with name class
// and give me the paragraph inside it
jQuery( '#citizen_overview_table td.name p' )

// This returns a jQueryfied ARRAY of DOM Elements

// Iterate over EACH entry
// where 'this' is the current DOM Element

.each( function() { 

// jQueryfied the current DOM Element into $t
var $t = jQuery(this), 

// Grab the parent next DOM Element item in the table
// This gives the next td entry
// which might contain the text "inside"
// and store this jQueryfied item into '$z'
$z = $t.parent().next(), 

// Is the next entry ($z) contains the word "inside"
// if so give back this DOM Element inside an array
inside = $z.filter( ':contains(inside)' )

// Is this array size not empty ?
// Therefore, inside is a boolean
// where inside = true,
// only if $z contains the text "inside"
.length > 0; 

// IF inside is TRUE THEN
if ( inside ) 
// Append the current player name trimmed into '$t'
// by converting the current DOM Element into text
// then applying TRIM
// then prefixing with ', @'
// and storing the result into '$t'
t += ', @' + jQuery.trim( jQuery(this).text() ); 

}); // Loop next "each" DOM element

// Remove the ", " prefix from the first loop
// and output the result to the DEV console
console.log( t.substring(2) );

New FIXED range script with AP / KM support

  • KM uses Pythagorean of (x,y)
  • AP uses sum of absolute values of (x,y)

11 AP ranges was added in this version useful for WT5 scavenging.

Works in MapViewer and FDTD

Just open Chrome Console (F12) and paste content below.

function getZones() {
 var $ = window.jQuery,
 j = "_",
 p = "#zone" + j,
 q = p.substr(1),
 css = "ul > li[id^=" + q + "]";
 return $(css);
}


function getZoneXY(zones0, i0) {
 var zones = zones0 || getZones(),
 i = +i0 || 0,
 f = zones[i > 0 ? i : 0],
 id = (f && f.id) || '',
 xy = id.substr(5).split(/_/g),
 x = +xy[0] || 0,
 y = +xy[1] || 0,
 r = [x, y];
 return r;
}


function pyth(x, y, isAP) {
 x = Math.abs(+x || 0);
 y = Math.abs(+y || 0);
 return +Math.round(isAP ? (x + y) : Math.sqrt(x * x + y * y));
}

function drawRange(zones0, dist, color, isAP) {
 var zones = zones0 || getZones() || [],
 nbZones = (zones && zones.length) || 0,
 $ = window.jQuery,
 i = 0,
 x = 0,
 y = 0,
 xy, o,
 blc = "border-left-color",
 brc = "border-right-color",
 bbc = "border-bottom-color",
 btc = "border-top-color";

 if (dist > 0 && color) {
 for (i = 0; i < nbZones; ++i) {
 xy = getZoneXY(zones, i);
 if (xy) {
 x = xy[0];
 y = xy[1];

 if (pyth(x, y, isAP) === dist) {
 o = zones.eq(i);
 if (x < 0 && pyth(x - 1, y, isAP) !== dist) {
 o.css(blc, color);
 } else if (x > 0 && pyth(x + 1, y) !== dist) {
 o.css(brc, color);
 } else if (isAP) {
 if (x === 0) {
 o.css(blc, color);
 o.css(brc, color);
 } else if (x === (dist - 1)) {
 o.css(brc, color);
 }
 }

 if (y < 0 && pyth(x, y - 1, isAP) !== dist) {
 o.css(bbc, color);
 } else if (y > 0 && pyth(x, y + 1) !== dist) {
 o.css(btc, color);
 } else if (isAP) {
 if (y === 0) {
 o.css(bbc, color);
 o.css(btc, color);
 } else if (y === (dist - 1)) {
 o.css(btc, color);
 }
 }
 }
 }
 }
 }
}

 drawRange(zones0, 11, "yellow");
}

 drawRange(zones0, 1, "gray");
